┌─ RESOLVER CONFLICTO ───────────────────────────────────────┐
│                                                            │
│ Escenario:                                                │
│ ├─ User A modifica en Android (offline)                  │
│ ├─ User B actualiza en web (modificando mismo registro) │
│ └─ Vuelve la conexión en Android                         │
│                                                            │
│ Proceso:                                                  │
│ 1. SyncEngine.pullChanges() ← Trae de Supabase          │
│                                                            │
│ 2. Compara versiones:                                     │
│    Local.updated_at ≠ Remote.updated_at → CONFLICTO      │
│                                                            │
│ 3. Busca estrategia:                                      │
│    conflictResolution = 'remote'                         │
│                                                            │
│ 4. Aplica estrategia:                                     │
│    Descargar Remote (remote gana)                        │
│                                                            │
│ 5. Marca:                                                 │
│    _sync_status = 'synced'                               │
│                                                            │
│ 6. Notifica a UI (opcional)                              │
│    onConflict callback                                    │
│                                                            │
│ ✅ CONFLICTO RESUELTO                                     │
│                                                            │

┌─ SCHEMA GENERATOR ───────────────────────────────────────┐
│ Problema anterior: Schema duplicado en .ts + .js         │
│                   = Mantenimiento imposible               │
│                                                           │
│ Solución:                                                │
│ packages/lib/src/db/schema-generator.ts                 │
│   export const SCHEMA_DEFINITION = {                    │
│     ideas: {                                             │
│       columns: [...]  ← Fuente única                    │
│       indexes: [...]                                     │
│     }                                                     │
│   }                                                       │
│                                                           │
│   Genera automáticamente:                                │
│   - SQL (CREATE TABLE, CREATE INDEX)                    │
│   - TypeScript (para otros idiomas)                     │
│   - Migraciones                                          │
│                                                           │
│ Resultado: ✅ Una sola lugar, siempre sync              │
└───────────────────────────────────────────────────────────┘

┌─ MIGRATIONS SYSTEM ──────────────────────────────────────┐
│ Problema anterior: Cambios en schema = Apps rotas        │
│                                                           │
│ Solución:                                                │
│ packages/lib/src/db/migrations.ts                       │
│   const ALL_MIGRATIONS = [                              │
│     { version: 1, up: () => CREATE TABLE ... },        │
│     { version: 2, up: () => ALTER TABLE ADD ... },     │
│     { version: 3, up: () => ... }                       │
│   ]                                                       │
│                                                           │
│   await applyMigrations(runSQL, ALL_MIGRATIONS)         │
│   → Detecta versión actual                              │
│   → Aplica solo las pendientes                          │
│   → Registra en tabla _migrations                       │
│                                                           │
│ Resultado: ✅ Upgrades seguros, sin downtime            │
└───────────────────────────────────────────────────────────┘

┌─ LOCAL DB ADAPTER ───────────────────────────────────────┐
│ Desacoplamiento: Lógica ≠ Implementación                │
│                                                           │
│ interface LocalDatabaseAdapter {                        │
│   query<T>(sql, params): Promise<T[]>                  │
│   execute(sql, params): Promise<{ changes }>           │
│   transaction<T>(callback): Promise<T>                 │
│   close(): Promise<void>                               │
│   getInfo(): Promise<{ dbName; version; ready }>       │
│ }                                                         │
│                                                           │
│ Implementaciones:                                        │
│ ├─ CapacitorSQLiteAdapter (Android/iOS)               │
│ └─ SqlJsAdapter (Web, futuro)                          │
│                                                           │
│ Resultado: ✅ Multi-plataforma, fácil de testear       │
└───────────────────────────────────────────────────────────┘

┌─ SYNC ENGINE ────────────────────────────────────────────┐
│ Sincronización offline-first                             │
│                                                           │
│ Fases:                                                   │
│ 1. pushChanges()   ← Enviar cambios locales             │
│ 2. pullChanges()   ← Traer cambios remotos              │
│ 3. resolveConflicts() ← Merge automático                │
│                                                           │
│ Características:                                         │
│ ├─ Cola de cambios (_sync_queue)                        │
│ ├─ Reintentos automáticos                              │
│ ├─ Estrategias conflictos (local/remote/manual)         │
│ ├─ Timestamps de sync                                   │
│ └─ Status tracking (_sync_status)                       │
│                                                           │
│ Resultado: ✅ Sincronización confiable                  │
└───────────────────────────────────────────────────────────┘

┌─ DUAL DATABASE MANAGER ──────────────────────────────────┐
│ Orquesta operaciones entre local y remote                │
│                                                           │
│ Modos:                                                   │
│ ├─ offline-first: Lee local, escribe local, sync async  │
│ ├─ remote-first: Lee remote + fallback, sync sync       │
│ └─ hybrid: Auto-detecta según conectividad             │
│                                                           │
│ Inteligencia:                                            │
│ write(table, data)                                       │
│   → Si offline-first: escribe local inmediato           │
│   → Si remote-first: intenta remoto, fallback local     │
│   → Si hybrid: detecta y elige                          │
│                                                           │
│ read(table)                                              │
│   → Si offline-first: lee local (rápido)                │
│   → Si remote-first: lee remoto con fallback            │
│   → Si hybrid: elige según conectividad                 │
│                                                           │
│ Resultado: ✅ Modo automático transparente              │
└───────────────────────────────────────────────────────────┘
